import groovy.json.JsonSlurper
pipeline {
   agent { label 'master' }
   stages {
       stage('read') {
           steps {
               script {
				   def jsonSlurper = new JsonSlurper()
                   def data = readFile(file: 'listExecutionPlans.json')
				   def config = jsonSlurper.parseText(data)
				   def duration = config.executionTime.millis
                   println(duration)
               }
           }
       }
   }
}
